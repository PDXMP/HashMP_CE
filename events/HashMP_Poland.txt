	id = flavor_pol.29
	title = flavor_pol.29.t
	desc = flavor_pol.29.d
	is_triggered_only = yes
	picture = SEJM_eventPicture
	

	option = {
		name = flavor_pol.29.a

			NOT = { 
				has_country_flag = purged_sejm 
			}
		}
		custom_tooltip = flavor_lit.bop.minus1
		add_estate_loyalty = {
			estate = estate_lithuanian_nobility
			loyalty = -10
		}
		add_estate_loyalty = {
			estate = estate_polish_nobility
			loyalty = 10
		}
		if = {
			limit = {
				tag = POL

					which = balanceOfPower
					value = 1
				}
			}
		}
		define_ruler = {
			dynasty = original_dynasty
			claim = 100
			male = yes
			min_age = 18
		}
		set_ruler_flag = polish_ruler
	}
	option = {
		name = flavor_pol.29.b
		trigger = { 
			NOT = { 
				OR ={
					has_country_flag = purged_sejm
					has_country_flag = purged_bajorais
				}
			}
		}
		define_ruler = {
			dynasty = original_dynasty

			min_age = 18
		}
		set_ruler_flag = compromise_ruler

		}
		set_ruler_flag = lithuanian_ruler

		}
