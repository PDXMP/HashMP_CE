country_event = {
	id = HashMP_Poland.107
	title = HashMP_Poland.107.t
	desc = HashMP_Poland.107.d
	picture = SEJM_eventPicture

	is_triggered_only = yes

	trigger = {
		has_reform = h_reform_elective_monarchy
	}

	option = {
		name = HashMP_Poland.107.a
		custom_tooltip = HashMP_Lithuanian_Influence_Tooltip_1

		trigger = { 
			NOT = { 
				has_country_flag = purged_sejm 
			}
		}

		add_estate_loyalty = {
			estate = estate_lithuanian_nobility
			loyalty = -10
		}
	
		add_estate_loyalty = {
			estate = estate_polish_nobility
			loyalty = 10
		}
	
		if = {
			limit = {
				tag = POL
			}
			hidden_effect = {
				subtract_variable = {
					which = h_polandlithuania_balance_of_power
					value = 1
				}
			}
		}

		define_ruler = {
			dynasty = original_dynasty
			claim = 100
			male = yes
			min_age = 18
			adm = 1
			dip = 1
			mil = 1
		}

		hidden_effect = { set_ruler_flag = polish_ruler }
	}
	option = {
		name = HashMP_Poland.107.b
	
		trigger = { 
			NOT = { 
				OR ={
					has_country_flag = purged_sejm
					has_country_flag = purged_bajorais
				}
			}
		}
	
		define_ruler = {
			dynasty = original_dynasty

				}
			}
		}

		define_ruler = {
			dynasty = original_dynasty
			claim = 100
			male = yes
			min_age = 20
			adm = 1
			dip = 1
			mil = 1
		}

		hidden_effect = { set_ruler_flag = lithuanian_ruler }
	}
	
	option = {
		name = HashMP_Poland.107.e
		custom_tooltip =HashMP_Polish_Influence_Tooltip_1

		add_estate_loyalty = {
			estate = estate_lithuanian_nobility
