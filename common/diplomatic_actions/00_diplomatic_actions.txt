# INSTRUCTIONS:
#
# condition				A diplomatic action can have any number of condition blocks, each with its own
#						tooltip, potential and allow section
#
# 	tooltip					Sets a custom text string similar to the hardcoded limits
# 							If no tooltip is scripted, the tooltip for the actual trigger will be shown
#							Note that the custom tooltip is only shown if the allow trigger is NOT met
#
# 	potential				Determines if the trigger is applicable or not
# 	allow					Determines if the action is valid or not
#
# effect				A diplomatic action can only have one effect block

# ROOT					actor
# FROM					target

declarewar = {
	condition = {
		potential = { 
			ROOT = { has_reform = h_reform_native_empire }
		}
		allow = { 
			always = no
		}
	}
	condition = {
		tooltip = h_tooltip_commanderywar
		potential = {
			ROOT = { has_reform = h_reform_commandery }
		}
		allow = {
			always = no
		}
	}
	condition = {
		tooltip = h_tooltip_cn_declarewar
		potential = {
			is_colonial_nation = yes
		}
		allow = {
			FROM = { same_continent = ROOT }
		}
	}
	effect = {
		if = {
			limit = {
				ROOT = { tag = DAN }
				FROM = { tag = NOR }
				has_country_flag = h_restore_the_union
				OR = {
					has_country_modifier = h_danish_defeat
					has_country_modifier = h_danish_defeat_ii
				}

				NOT = {
					OR = {
						has_country_modifier = h_danish_defeat_iii
						has_country_modifier = h_danish_defeat_iv
					}
				}
			}

			ROOT = { add_stability = 2 }
		}

		if = {
			limit = {
				ROOT = { tag = DAN }
				FROM = { tag = NOR }
				has_country_flag = h_restore_the_union
				OR = {
					has_country_modifier = h_danish_defeat_iii
					has_country_modifier = h_danish_defeat_iv
				}

				NOT = {
					OR = {
						has_country_modifier = h_danish_defeat
						has_country_modifier = h_danish_defeat_ii
					}
				}
			}

			ROOT = { add_stability = 1 }
		}
	}
}
